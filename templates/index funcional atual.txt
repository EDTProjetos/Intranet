  <script>
    document.addEventListener("DOMContentLoaded", function () {
      let loggedIn = false;
      let requestedSection = null;

      const headerTitle = document.getElementById("headerTitle");
      const logo = document.querySelector(".logo");
      const googleDriveLink = document.getElementById("googleDriveLink");
      const planejamentoBtn = document.getElementById("planejamentoBtn");
      const planejamentoSubmenu = document.getElementById("planejamentoSubmenu");
      const planejamentoBIBtn = document.getElementById("planejamentoBIBtn");
      const planejamentoPerfilBtn = document.getElementById("planejamentoPerfilBtn");

      const toggleSidebarBtn = document.getElementById("toggleSidebarBtn");
      const sidebar = document.querySelector(".sidebar");

      const loginMenu = document.getElementById("loginMenu");
      const loginBtn = document.getElementById("loginBtn");
      const usernameInput = document.getElementById("username");
      const passwordInput = document.getElementById("password");
      const loginError = document.getElementById("loginError");

      const profileMenu = document.getElementById("profileMenu");
      const generateProfileBtn = document.getElementById("generateProfileBtn");
      const executionTime = document.getElementById("executionTime");
      const successMessage = document.getElementById("successMessage");

      const driveSection = document.getElementById("driveSection");
      const planejamentoSection = document.getElementById("planejamentoSection");

      // ----------------- SLIDER -----------------
      const slider = document.getElementById("slider");
      const slides = slider.querySelectorAll(".slide");
      const arrowLeft = document.getElementById("arrowLeft");
      const arrowRight = document.getElementById("arrowRight");
      const progressFill = document.getElementById("progressFill");
      const pauseSlideBtn = document.getElementById("pauseSlideBtn");

      let currentSlideIndex = 0;
      let slideInterval = 6000; // Tempo de cada imagem (6s)
      let slideTimer = null;
      let progressTimer = null;
      let progress = 0;
      let isPaused = false; // Controle para pausar/retomar

      function showSlide(index) {
        slides.forEach((slide, i) => {
          slide.classList.toggle('active', i === index);
        });
      }

      function nextSlide() {
        currentSlideIndex = (currentSlideIndex + 1) % slides.length;
        showSlide(currentSlideIndex);
        resetProgress();
      }

      function prevSlide() {
        currentSlideIndex = (currentSlideIndex - 1 + slides.length) % slides.length;
        showSlide(currentSlideIndex);
        resetProgress();
      }

      function startSlider() {
        // Timer de troca automática
        slideTimer = setInterval(nextSlide, slideInterval);

        // Timer da barra de progresso (atualiza a cada 100ms)
        progressTimer = setInterval(() => {
          progress += (100 / (slideInterval / 100));
          if (progress >= 100) {
            progress = 100;
          }
          progressFill.style.width = progress + '%';
        }, 100);
      }

      function stopSlider() {
        clearInterval(slideTimer);
        clearInterval(progressTimer);
      }

      function resetProgress() {
        progress = 0;
        progressFill.style.width = '0%';
        stopSlider();
        if (!isPaused) {
          startSlider();
        }
      }

      // Botão de pausar/retomar
      pauseSlideBtn.addEventListener("click", function() {
        if (!isPaused) {
          // Pausar
          isPaused = true;
          stopSlider();
          pauseSlideBtn.innerText = "Retomar";
        } else {
          // Retomar
          isPaused = false;
          startSlider();
          pauseSlideBtn.innerText = "Pausar";
        }
      });

      // Inicia o slider
      showSlide(currentSlideIndex);
      startSlider();

      // Eventos das setas
      arrowLeft.addEventListener("click", prevSlide);
      arrowRight.addEventListener("click", nextSlide);

      // ----------------- FIM SLIDER -----------------

      // Exibe o slider somente na página inicial (Intranet EDT)
      function showSliderDiv() {
        if (slider) {
          slider.style.display = "block";
        }
      }
      function hideSliderDiv() {
        if (slider) {
          slider.style.display = "none";
        }
      }

      // Inicialmente, como estamos na página inicial, mostramos o slider
      showSliderDiv();
      hideAllMenus();
      updateHeaderTitle("Intranet EDT");

      // Função para esconder todos os menus (exceto o slider)
      function hideAllMenus() {
        loginMenu.style.display = "none";
        profileMenu.style.display = "none";
        driveSection.style.display = "none";
        planejamentoSection.style.display = "none";
      }

      // Atualiza o título do cabeçalho
      function updateHeaderTitle(text) {
        headerTitle.innerText = text;
      }

      // Esconde a sidebar
      function hideSidebar() {
        sidebar.style.transform = "translateX(-100%)";
        toggleSidebarBtn.innerText = "Mostrar Sidebar";
      }

      // Clique na logo: volta para a página inicial
      logo.addEventListener("click", function () {
        hideAllMenus();
        updateHeaderTitle("Intranet EDT");
        showSliderDiv();
        requestedSection = null;
      });

      // Ao clicar em qualquer botão da sidebar (exceto a logo), esconde o slider
      googleDriveLink.addEventListener("click", function () {
        hideAllMenus();
        updateHeaderTitle(this.innerText);
        driveSection.style.display = "block";
        hideSliderDiv();
        requestedSection = null;
      });

      planejamentoBtn.addEventListener("click", function() {
        if (planejamentoSubmenu.style.display === "none" || planejamentoSubmenu.style.display === "") {
          planejamentoSubmenu.style.display = "flex";
          planejamentoSubmenu.style.flexDirection = "column";
        } else {
          planejamentoSubmenu.style.display = "none";
        }
        hideSliderDiv();
      });

      planejamentoBIBtn.addEventListener("click", function() {
        hideAllMenus();
        if (loggedIn) {
          planejamentoSection.style.display = "block";
          updateHeaderTitle("Planejamento BI");
          hideSidebar();
        } else {
          loginError.style.display = "none";
          usernameInput.value = "";
          passwordInput.value = "";
          loginMenu.style.display = "block";
          requestedSection = "planejamentoBI";
          updateHeaderTitle("Login - Planejamento BI");
        }
        hideSliderDiv();
      });

      planejamentoPerfilBtn.addEventListener("click", function() {
        hideAllMenus();
        if (loggedIn) {
          profileMenu.style.display = "block";
          updateHeaderTitle("Geração de perfil");
          hideSidebar();
        } else {
          loginError.style.display = "none";
          usernameInput.value = "";
          passwordInput.value = "";
          loginMenu.style.display = "block";
          requestedSection = "planejamentoPerfil";
          updateHeaderTitle("Login - Geração de perfil");
        }
        hideSliderDiv();
      });

      loginBtn.addEventListener("click", function () {
        if (usernameInput.value === "admin" && passwordInput.value === "admin") {
          loggedIn = true;
          loginMenu.style.display = "none";

          if (requestedSection === "planejamentoBI") {
            planejamentoSection.style.display = "block";
            updateHeaderTitle("Planejamento BI");
            hideSidebar();
          } else if (requestedSection === "planejamentoPerfil") {
            profileMenu.style.display = "block";
            updateHeaderTitle("Geração de perfil");
            hideSidebar();
          } else {
            updateHeaderTitle("Intranet EDT");
          }
        } else {
          loginError.style.display = "block";
        }
      });

      generateProfileBtn.addEventListener("click", function () {
        generateProfileBtn.disabled = true;
        generateProfileBtn.style.backgroundColor = "#A9A9A9";
        generateProfileBtn.innerText = "Gerando...";

        successMessage.style.display = "none";
        executionTime.style.display = "none";

        fetch("/executar-script", { method: "POST" })
          .then(response => response.json())
          .then(data => {
            console.log(data.message);
            const currentTime = new Date().toLocaleString();
            localStorage.setItem("lastExecutionTime", currentTime);
            executionTime.innerText = `Última execução: ${currentTime}`;
            executionTime.style.display = "block";

            setTimeout(() => {
              successMessage.style.display = "block";
              generateProfileBtn.disabled = false;
              generateProfileBtn.style.backgroundColor = "#009688";
              generateProfileBtn.innerText = "Gerar Perfil";

              setTimeout(() => {
                successMessage.style.display = "none";
              }, 3000);
            }, 10000);
          })
          .catch(error => {
            console.error("Erro ao executar o script:", error);
            alert("Ocorreu um erro ao gerar o perfil.");
            generateProfileBtn.disabled = false;
            generateProfileBtn.style.backgroundColor = "#009688";
            generateProfileBtn.innerText = "Gerar Perfil";
          });
      });

      // Outros botões da sidebar
      const otherButtons = document.querySelectorAll(
        ".sidebar .button:not(#googleDriveLink):not(#planejamentoBtn):not(#planejamentoBIBtn):not(#planejamentoPerfilBtn)"
      );
      otherButtons.forEach(btn => {
        btn.addEventListener("click", function () {
          hideAllMenus();
          updateHeaderTitle(this.innerText);
          requestedSection = null;
          hideSliderDiv();
        });
      });

      toggleSidebarBtn.addEventListener("click", function () {
        if (sidebar.style.transform === "translateX(-100%)") {
          sidebar.style.transform = "translateX(0)";
          toggleSidebarBtn.innerText = "Ocultar Sidebar";
        } else {
          sidebar.style.transform = "translateX(-100%)";
          toggleSidebarBtn.innerText = "Mostrar Sidebar";
        }
      });

      // ------------------- RELATÓRIOS DE BI (HORIZONTAL) -------------------
      const biReportsMenu = document.getElementById("biReportsMenu");
      const reportButtons = biReportsMenu.querySelectorAll(".button");
      const planejamentoFrame = document.getElementById("planejamentoFrame");
      let currentReportIndex = 0;

      function updateIframeFromIndex() {
        if (reportButtons.length > 0) {
          const btn = reportButtons[currentReportIndex];
          const newSrc = btn.getAttribute("data-src");
          planejamentoFrame.src = newSrc;
          // Atualiza o título da seção para o texto do botão clicado
          document.querySelector("#planejamentoSection h2").innerText = btn.innerText;
        }
      }

      reportButtons.forEach((btn, index) => {
        btn.addEventListener("click", () => {
          currentReportIndex = index;
          updateIframeFromIndex();
        });
      });

      const prevReport = document.getElementById("prevReport");
      const nextReport = document.getElementById("nextReport");

      prevReport.addEventListener("click", function() {
        if (currentReportIndex > 0) {
          currentReportIndex--;
          updateIframeFromIndex();
        }
      });

      nextReport.addEventListener("click", function() {
        if (currentReportIndex < reportButtons.length - 1) {
          currentReportIndex++;
          updateIframeFromIndex();
        }
      });

      // ------------------- TRANSPARÊNCIA DO CABEÇALHO -------------------
      window.addEventListener("scroll", function() {
        // Se estivermos na página de Planejamento BI, ao dar scroll o cabeçalho fica com opacidade reduzida
        if (planejamentoSection && planejamentoSection.style.display !== "none") {
          if (window.scrollY > 10) {
            headerTitle.style.opacity = 0;
          } else {
            headerTitle.style.opacity = 1;
          }
        } else {
          headerTitle.style.opacity = 1;
        }
      });
    });
  </script>
